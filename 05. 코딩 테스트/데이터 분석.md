# 1차 풀이

import java.util.*;

class Solution {
    public int[][] solution(int[][] data, String ext, int val_ext, String sort_by) {
        return Arrays.stream(data)
            .filter(d -> d[getIdxByCategory(ext)] < val_ext)
            .sorted((d1, d2) -> d1[getIdxByCategory(sort_by)] - d2[getIdxByCategory(sort_by)])
            .toArray(int[][]::new);
    }
    
    private int getIdxByCategory(String value) {
        if (value.equals("code")) return 0;
        if (value.equals("date")) return 1;
        if (value.equals("maximum")) return 2;
        return 3;
    }
}
