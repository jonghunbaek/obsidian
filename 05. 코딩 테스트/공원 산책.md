# 1차 풀이
```java
// 완전 탐색

class Solution {
    public int[] solution(String[] park, String[] routes) {
        char[][] map = new char[park.length][park[0].length()];
        int y = 0;
        int x = 0;
        for (int i = 0; i < park.length; i++) {
            map[i] = park[i].toCharArray();
            for (int j = 0; j < map[i].length; j++) {
                if (map[i][j] == 'S') {
                    y = i;
                    x = j;
                }
            }
        }
             
        
        for (String route : routes) {
            int[] direction = getVectorByRoute(route.split(" ")[0]);
            int distance = Integer.parseInt(route.split(" ")[1]);
            
            int ny = y;
            int nx = x;
            boolean isMovable = true;
            for (int i = 1; i <= distance; i++) {
                ny += direction[0];
                nx += direction[1];
                if (nx < 0 || nx >= map[0].length || ny < 0 || ny >= map.length || map[ny][nx] == 'X') isMovable = false;
            }
            
            if (!isMovable) continue;
            
            y = ny;
            x = nx;
        }
        
        return new int[] {y, x};
    }
    
    private int[] getVectorByRoute(String direction) {
        if ("E".equals(direction)) return new int[] {0, 1};
        if ("W".equals(direction)) return new int[] {0, -1};
        if ("S".equals(direction)) return new int[] {1, 0};
        return new int[] {-1, 0};
    }
}
```
