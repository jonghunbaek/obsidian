# 1차 풀이
```java
class Solution {
    public int solution(String t, String p) {
        int size = p.length();
        long origin = Long.parseLong(p);
        int idx = 0;
        int answer = 0;
        while (idx <= t.length() - size) {
            long target = Long.parseLong(t.substring(idx, idx + size));
            if (target <= origin) answer++;
            idx++;
        }
        return answer;
    }
}
```
