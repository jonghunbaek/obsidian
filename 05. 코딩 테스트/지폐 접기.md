# 1차 풀이
```java
import java.util.*;

class Solution {
    public int solution(int[] wallet, int[] bill) {
        Arrays.sort(wallet);
        Arrays.sort(bill);
        
        int answer = 0;
        int small = bill[0];
        int large = bill[1];
        while (small > wallet[0] || large > wallet[1]) {
            large /= 2;
            
            int temp = 0;
            if (large < small) {
                temp = small;
                small = large;
                large = temp;
            }
            
            answer++;
        }
        
        return answer;
    }
}
```

# 2차 풀이
```java
class Solution {
    public int solution(int[] wallet, int[] bill) {
        int answer = 0;
        while (min(wallet) < min(bill) || max(wallet) < max(bill)) {
            if (bill[0] > bill[1]) {
                bill[0] /= 2;
            } else {
                bill[1] /= 2;
            }
            
            answer++;
        }
        
        return answer;
    }
    
    private int min(int[] values) {
        return Math.min(values[0], values[1]);
    }
    
    private int max(int[] values) {
        return Math.max(values[0], values[1]);
    }
}
```
