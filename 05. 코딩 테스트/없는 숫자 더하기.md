class Solution {
    public int solution(int[] numbers) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < 10; i++) {
            set.add(i);
        }
        
        for (int number : numbers) {
            if (set.contains(number)) {
                set.remove(number);
            }
        }
        
        return set.stream()
            .mapToInt(Integer::intValue)
            .sum();
    }
}
