# 1차 풀이
```java
// 1. 2차원 배열의 최대 크기는 1000 * 1000
// 2. skill의 최대 길이는 250,000
// 3. 즉, 완탐으로 수행하면 무조건 시간초과
// 4. 다만, DP로 풀 경우 해답이 보이지 않으니 우선 완탐으로 풀고 memoization을 할 수 있는 부분을 찾기

class Solution {
    public int solution(int[][] board, int[][] skill) {
        for (int[] s : skill) {        
            dealOrHeal(s[0] == 2, new int[] {s[1], s[2], s[3], s[4]}, s[5], board);
        }
        
        int result = 0;
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                if (board[i][j] >= 1) result++;
            }
        }
        
        return result;
    }
    
    private void dealOrHeal(boolean isHealing, int[] range, int value, int[][] board) {
        for (int i = range[0]; i <= range[2]; i++) {
            for (int j = range[1]; j <= range[3]; j++) {
                if (isHealing) { 
                    board[i][j] += value;
                } else {
                    board[i][j] -= value;
                }
            }
        }
    }
}
```
문제의 조건들을 통해 해당 문제가 dp 유형의 문제라고 판단.
단순 정확성은 모두 통과지만 효율성에서 통과 못함

정확히는 dp가 아닌 [[누적합, 차분배열]]으로 접근해야 함
```java
class Solution {
    public int solution(int[][] board, int[][] skill) {
        int n = board.length;
        int m = board[0].length;
        
        // 차분 배열 (크기를 1씩 크게)
        int[][] diff = new int[n + 1][m + 1];
        
        // 1. 각 스킬마다 차분 배열에 표시만 함
        for (int[] s : skill) {
            int type = s[0];
            int r1 = s[1], c1 = s[2];
            int r2 = s[3], c2 = s[4];
            int degree = s[5] * (type == 1 ? -1 : 1);
            
            // 구간 [r1,c1] ~ [r2,c2]에 degree를 더하는 것을
            // 4개 모서리에 표시
            diff[r1][c1] += degree;
            diff[r1][c2 + 1] -= degree;
            diff[r2 + 1][c1] -= degree;
            diff[r2 + 1][c2 + 1] += degree;
        }
        
        // 2. 누적합으로 실제 변화량 계산 (가로 방향)
        for (int i = 0; i < n; i++) {
            for (int j = 1; j < m; j++) {
                diff[i][j] += diff[i][j - 1];
            }
        }
        
        // 3. 누적합으로 실제 변화량 계산 (세로 방향)
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                diff[i][j] += diff[i - 1][j];
            }
        }
        
        // 4. 원본 board에 변화량 적용하고 카운트
        int result = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (board[i][j] + diff[i][j] > 0) {
                    result++;
                }
            }
        }
        
        return result;
    }
}
```

        for (int[] s : skill) {
            int type = s[0];
            int r1 = s[1];
            int c1 = s[2];
            int r2 = s[3];
            int c2 = s[4];
            int degree = type == 1 ? -1 * s[5] : s[5];
            
            prefixSum[r1][c1] += degree;
            prefixSum[r1][c2 + 1] -= degree;
            prefixSum[r2 + 1][c1] -= degree;
            prefixSum[r2 + 1][c2 + 1] += degree;
        }
        
        // 세로 방향 누적합
        for (int i = 1; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                prefixSum[i][j] += prefixSum[i - 1][j];
            }
        }
        
        // 가로 방향 누적합
        for (int i = 0; i < board.length; i++) {
            for (int j = 1; j < board[i].length; j++) {
                prefixSum[i][j] += prefixSum[i][j - 1];

        }
        
        int result = 0;
        for (int i = 0; i < board.length; i++) {

}
