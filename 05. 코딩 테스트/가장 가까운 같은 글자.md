# 1차 풀이
```java
public int[] solution(String s) {  
    int[] results = new int[s.length()];  
    Map<Character, Integer> indexByChar = new HashMap<>();  
    for (int i = 0; i < s.length(); i++) {  
        char c = s.charAt(i);  
        Integer index = indexByChar.getOrDefault(c, -1);  
        results[i] = index == -1 ? -1 : i - index;  
  
        indexByChar.put(c, i);  
    }  
  
    return results;  
}
```
