# 1차 풀이
```java
class Solution {
    public String solution(String s, int n) {
        int upperMax = 'Z';
        int upperMin = 'A';
        int lowerMax = 'z';
        int lowerMin = 'a';
        char[] chars = s.toCharArray();
        char[] results = new char[chars.length];
        for (int i = 0; i < chars.length; i++) {
            if (Character.isWhitespace(chars[i])) {
                results[i] = ' ';
                continue;
            }

            if (Character.isUpperCase(chars[i])) {
                int idx = chars[i];
                if (idx + n > upperMax) {
                    int val = idx + n - upperMax - 1;
                    results[i] = (char) (upperMin + val);
                } else {
                    results[i] = (char)(idx + n);
                }
            } else {
                int idx = chars[i];
                if (idx + n > lowerMax) {
                    int val = idx + n - lowerMax - 1;
                    results[i] = (char) (lowerMin + val);
                } else {
                    results[i] = (char)(idx + n);
                }
            }
        }
        return String.valueOf(results);
    }
}

class Solution {
    // 문자열을 순회한다.
    // 순회하며 각 문자를 지정된 거리 n만큼 밀어낸다.
    // z -> a, Z -> A
    public String solution(String s, int n) {
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < s.length(); i++) {
            char value = s.charAt(i);
            if ('a' <= value && value <= 'z') {
                int idx = (value - 'a' + n) % 26;
                value = (char) ('a' + idx);
            }
            
            if ('A' <= value && value <= 'Z') {
                int idx = (value - 'A' + n) % 26;
                value = (char) ('A' + idx);
            }
            
            sb.append(value);
        }
        
        return sb.toString();
    }
}

class Solution {
    public String solution(String s, int n) {
        char[] arr = s.toCharArray();
        for (int i=0; i<arr.length; i++) {
            if (!Character.isAlphabetic(arr[i])) {
                continue;
            }
            
            int offset = Character.isUpperCase(arr[i]) ? 'A' : 'a';
            int nextPosition = (arr[i] - offset + n) % ('Z' - 'A' + 1);
            
            arr[i] = (char) (offset + nextPosition);
        }
        
        return new String(arr);
    }
}
```

# 2차 풀이
```java
class Solution {
    public String solution(String s, int n) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (Character.isWhitespace(s.charAt(i))) {
                sb.append(s.charAt(i));
                continue;
            }
            
            int value = s.charAt(i) + n;
            if (Character.isUpperCase(s.charAt(i))) {
                if (value > 'Z') {
                    sb.append((char)('A' + (value - 'Z' - 1)));
                } else {
                    sb.append((char) value);
                }
            } else {
                if (value > 'z') {
                    sb.append((char)('a' + (value - 'z' - 1)));
                } else {
                    sb.append((char) value);
                }
            }
        }
        
        return sb.toString();
    }
}
```

```java
class Solution {

        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) {
            if (!Character.isAlphabetic(c)) {
                sb.append(c);
                continue;
            }

            int offset = Character.isUpperCase(c) ? 'A' : 'a';
            char nextChar = (char)(((c - offset + n) % 26) + offset);
            sb.append(nextChar);
        }
        
        return sb.toString();
    }
}
```
