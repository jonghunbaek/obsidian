# 1차 풀이
```java
// 그리디, 정렬
// 가장 낮은 점수가 최소한의 박스에 포함되게

import java.util.*;

class Solution {
    public int solution(int k, int m, int[] score) {
        int[] scores = Arrays.stream(score).boxed()
            .sorted(Collections.reverseOrder())
            .limit((score.length / m) * m)
            .mapToInt(Integer::intValue)
            .toArray();
        
        int sum = 0;
        for (int i = 0; i < scores.length; i += m) {
            sum += scores[i + m - 1] * m;
        }
        
        return sum;
    }
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public int solution(int k, int m, int[] score) {
        score = Arrays.stream(score).boxed()
            .sorted((i1, i2) -> i2 - i1)
            .mapToInt(Integer::intValue)
            .toArray();
        
        int sum = 0;
        for (int i = 0; i < score.length; i += m) {
            if (i + m - 1 >= score.length) break;
            sum += score[i + m - 1] * m;
        }
        
        return sum;
    }
}
```
