public int solution(int[] d, int budget) {  
    return plan(0, 0, 0, d, budget);  
}  
  
private int plan(int idx, int sum, int count, int[] d, int budget) {  
    if (sum > budget) {  
        return count - 1;  
    }  
  
    if (count == d.length || idx == d.length) {  
        return count;  
    }  
  
    int max = 0;  
    for (int i = idx; i < d.length; i++) {  
        int value = plan(i + 1, sum + d[i], count + 1, d, budget);  
        max = Math.max(max, value);  
        if (max == d.length) {  
            break;  
        }  
    }  
  
    return max;  
}
