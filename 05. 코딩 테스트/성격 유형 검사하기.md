# 1차 풀이
```java
// 구현, 해시
// 성격 유형, 동의에 따른 점수 미리 설정
// survey와 choices를 함께 순회하며, 점수 계산
import java.util.*;

class Solution {
    public String solution(String[] survey, int[] choices) {
        int[] scores = createScores();
        Map<String, Integer> scoreByType = new HashMap<>();
        for (int i = 0; i < survey.length; i++) {
            int choice = choices[i];
            int score = scores[choice - 1];
            String[] types = survey[i].split("");
            
            if (choice == 4) continue;            
            if (choice < 4) {
                scoreByType.put(types[0], scoreByType.getOrDefault(types[0], 0) + score);
            } else {
                scoreByType.put(types[1], scoreByType.getOrDefault(types[1], 0) + score);
            }
        }
        
        StringBuilder results = new StringBuilder();
        String firstResult = compare(scoreByType.getOrDefault("R", 0), scoreByType.getOrDefault("T", 0), "RT");
        String secondResult = compare(scoreByType.getOrDefault("C", 0), scoreByType.getOrDefault("F", 0), "CF");
        String thirdResult = compare(scoreByType.getOrDefault("J", 0), scoreByType.getOrDefault("M", 0), "JM");
        String fourthResult = compare(scoreByType.getOrDefault("A", 0), scoreByType.getOrDefault("N", 0), "AN");
        
        results.append(firstResult);
        results.append(secondResult);
        results.append(thirdResult);
        results.append(fourthResult);
        
        return results.toString();
    }
    
    private int[] createScores() {
        int[] scores = new int[7];
        int base = 3;
        int offset = 1;
        int value = 1;
        while (offset < 4) {
            scores[base + offset] = value;
            scores[base - offset] = value;
            offset++;
            value++;
        }
        
        return scores;
    }
    
    private String compare(int value1, int value2, String target) {
        if (value1 >= value2) return target.split("")[0];
        return target.split("")[1];
    }
    
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public String solution(String[] survey, int[] choices) {
        int[] scores = {3, 2, 1, 0, 1, 2, 3};
        String[] categories = {"RT", "CF", "JM", "AN"};        
        Map<String, Integer> myScores = new HashMap<>();
        for (int i = 0; i < choices.length; i++) {
            int choice = choices[i];
            if (choice == 4) continue;
            
            if (choice > 4) {
                myScores.put(survey[i].split("")[1], myScores.getOrDefault(survey[i].split("")[1], 0) + scores[choice - 1]);
            } else {
                myScores.put(survey[i].split("")[0], myScores.getOrDefault(survey[i].split("")[0], 0) + scores[choice - 1]);
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < categories.length; i++) {
            String[] types = categories[i].split("");
            int type1Score = myScores.getOrDefault(types[0], 0);
            int type2Score = myScores.getOrDefault(types[1], 0);
            
            if (type1Score >= type2Score) {
                sb.append(types[0]);
            } else {
                sb.append(types[1]);
            }
        }
        
        return sb.toString();
    }
}
```
