# 1차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(int[] answers) {
        int[] supoja1 = new int[] {1, 2, 3, 4, 5};
        int[] supoja2 = new int[] {2, 1, 2, 3, 2, 4, 2, 5};
        int[] supoja3 = new int[] {3, 3, 1, 1, 2, 2, 4, 4, 5, 5};
        
        int[] counts = new int[3];
        for (int i = 0; i < answers.length; i++) {
            int answer = answers[i];
            
            if (answer == supoja1[i % supoja1.length]) {
                counts[0]++;
            } 
            
            if (answer == supoja2[i % supoja2.length]) {
                counts[1]++;
            } 
            
            if (answer == supoja3[i % supoja3.length]) {
                counts[2]++;
            }
        }
        
        int max = -1;
        for (int i = 0; i < counts.length; i++) {
            max = Math.max(max, counts[i]);
        }
        
        List<Integer> results = new ArrayList<>();
        for (int i = 0; i < counts.length; i++) {
            if (max == counts[i]) {
                results.add(i + 1);
            }
        }
        
        return results.stream()
            .mapToInt(Integer::intValue)
            .toArray();
    }
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(int[] answers) {
        int[] student1 = new int[] {1, 2, 3, 4, 5};
        int[] student2 = new int[] {2, 1, 2, 3, 2, 4, 2, 5};
        int[] student3 = new int[] {3, 3, 1, 1, 2, 2, 4, 4, 5, 5};
        int[] count = new int[3];
        for(int i = 0; i < answers.length; i++) {
            int answer = answers[i];
            
            if (student1[i % 5] == answer) count[0]++;
            if (student2[i % 8] == answer) count[1]++;
            if (student3[i % 10] == answer) count[2]++;
        }
        
        List<Integer> results = new ArrayList<>();
        int max = Math.max(count[0], Math.max(count[1], count[2]));
        for (int i = 0; i < count.length; i++) {
            if (max == count[i]) {
                results.add(i + 1);
            }
        }
        
        return results.stream()
            .sorted()
            .mapToInt(Integer::intValue)
            .toArray();
    }
}
```
