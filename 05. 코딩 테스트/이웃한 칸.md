# 1차 풀이
```java
class Solution {
    public int solution(String[][] board, int h, int w) {
        int answer = 0;
        String color = board[h][w];
        int[] dx = new int[] {1, -1, 0, 0};
        int[] dy = new int[] {0, 0, 1, -1};
        for (int i = 0; i < 4; i++) {
            int nx = dx[i] + w;
            int ny = dy[i] + h;
            
            if (nx < 0 || nx >= board[0].length || ny < 0 || ny >= board.length || !board[ny][nx].equals(color)) continue;
            
            answer++;
        }
        return answer;
    }
}
```

```java
class Solution {
    public int solution(String[][] board, int h, int w) {
        int answer = 0;
        String color = board[h][w];
        int[] dx = new int[] {1, -1, 0, 0};
        int[] dy = new int[] {0, 0, 1, -1};
        for (int i = 0; i < 4; i++) {
            int nx = w + dx[i];
            int ny = h + dy[i];
            if (nx < 0 || nx >= board[h].length || ny < 0 || ny >= board.length || !board[ny][nx].equals(color)) continue;
            answer++;
        }
        return answer;
    }
}
```
