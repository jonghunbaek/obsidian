# 1차 풀이
```java
// 1. 단순 구현. 단, report의 최대 길이가 200,000임에 주의
// 2. id_list의 원소를 키로 신고된 횟수를 값으로 Map에 저장 Map<String, Integer> counts
// 3. id_list의 원소를 키로 신고한 유저의 이름을 Set타입의 값으로 Map에 저장 Map<String, Set<String>> set
// 4. set의 값을 필터링하여 몇번의 메일을 보내야할지 확인

import java.util.*;
import java.util.stream.*;

class Solution {
    public int[] solution(String[] id_list, String[] report, int k) {
        Map<String, Integer> counts = new HashMap<>();
        Map<String, Set<String>> set = Arrays.stream(id_list)
            .collect(Collectors.toMap(id -> id, id -> new HashSet<>()));
        
        for (String r : report) {
            String[] split = r.split(" ");
            String origin = split[0];
            String target = split[1];
            
            Set<String> ids = set.get(origin);
            int value = counts.getOrDefault(target, 0);
            if(!ids.contains(target)) {
                value++;
            }
            ids.add(target);
            set.put(origin, ids);
            counts.put(target, value);
        }
        
        int[] results = new int[id_list.length];
        for (int i = 0; i < id_list.length; i++) {
            Set<String> targets = set.get(id_list[i]);
            
            results[i] = (int) targets.stream()
                .filter(target -> counts.get(target) >= k)
                .count();
        }
        
        return results;
    }
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(String[] id_list, String[] report, int k) {
        Map<String, Set<String>> reports = new HashMap<>();
        Map<String, Integer> counts = new HashMap<>();
        for (String id : id_list) {
            reports.put(id, new HashSet<>());
        }
        
        for (String r : report) {
            String[] split = r.split(" ");
            String origin = split[0];
            String target = split[1];
            
            if (reports.get(origin).contains(target)) continue;
            Set<String> targets = reports.get(origin);
            targets.add(target);
            reports.put(origin, targets);
            counts.put(target, counts.getOrDefault(target, 0) + 1);
        }
        
        int[] results = new int[id_list.length];
        for (int i = 0; i < results.length; i++) {
            Set<String> targets = reports.get(id_list[i]);
            results[i] = (int) targets.stream()
                .filter(target -> counts.get(target) >= k)
                .count();
        }
        
        return results;
    }
}
```
