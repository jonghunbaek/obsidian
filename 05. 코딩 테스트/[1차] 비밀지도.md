# 1차 풀이
```java
// 구현
// 1. n번을 순회하며 arr1, arr2의 원소를 2진수 문자열로 변환
// 2. 2진수 문자열을 split
// 3. 순회하며 둘 중 하나가 #이면 #으로, 아니라면 공백으로 문자열 더하기
// 4. 문자열을 만들어 결과 배열에 저장

import java.util.*;

class Solution {
    public String[] solution(int n, int[] arr1, int[] arr2) {
        String[] results = new String[n];
        for (int i = 0; i < n; i++) {
            String[] binary1 = repeatZero(Integer.toString(arr1[i], 2), n).split("");
            String[] binary2 = repeatZero(Integer.toString(arr2[i], 2), n).split("");
            
            String result = "";
            for (int j = 0; j < n; j++) {
                if ("1".equals(binary1[j]) || "1".equals(binary2[j])) {
                    result += "#";
                } else {
                    result += " ";
                }
            }
            
            results[i] = result;
        }
        
        return results;
    }
    
    private String repeatZero(String target, int size) {
        if (target.length() == size) return target;
        
        return "0".repeat(size - target.length()) + target;
    }
}
```

        }
        
        return results;
    }
    
    private String repeatZero(String target, int n) {

        return "0".repeat(n - target.length()) + target;
    }
}
