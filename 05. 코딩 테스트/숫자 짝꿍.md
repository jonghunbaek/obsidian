# 1차 풀이
```java
// 문자열, 해시
// 1. x,y 각 문자열을 순회하며 0 ~ 9를 키로 몇 개가 나오는지 값으로 저장
// 2. 0 ~ 9를 순회하며 map에서 값을 꺼내 개수일치여부 확인해 일치하면 개수만큼 List에 추가
// 3. 해당 숫자를 내림차순 정렬해 문자로 만들어 반환
import java.util.*;
import java.util.stream.*;

class Solution {
    public String solution(String X, String Y) {
        Map<Character, Integer> xMap = new HashMap<>();
        Map<Character, Integer> yMap = new HashMap<>();
        for (int i = 0; i < X.length(); i++) {
            xMap.put(X.charAt(i), xMap.getOrDefault(X.charAt(i), 0) + 1);
        }
        
        for (int i = 0; i < Y.length(); i++) {
            yMap.put(Y.charAt(i), yMap.getOrDefault(Y.charAt(i), 0) + 1);
        }
        
        List<String> results = new ArrayList<>();
        for (int i = 9; i >= 0; i--) {
            char key = (char)('0' + i);
            int xCount = xMap.getOrDefault(key, -1);
            int yCount = yMap.getOrDefault(key, -1);
            
            if (xCount == -1 || yCount == -1) continue;
            results.add(String.valueOf(key).repeat(Math.min(xCount, yCount)));
        }
        
        if (results.isEmpty()) return "-1";
        
        String result = results.stream()
            .collect(Collectors.joining());
        
        return result.startsWith("0") ? "0" : result;
    }
}
```

# 2차 풀이

```java
class Solution {
    public String solution(String X, String Y) {
        int[] xCount = new int[10];
        int[] yCount = new int[10];
        for (char c : X.toCharArray()) {
            xCount[c - '0']++;
        }
        
        for (char c : Y.toCharArray()) {
            yCount[c - '0']++;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 9; i >= 0; i--) {
            sb.append(String.valueOf(i).repeat(Math.min(xCount[i], yCount[i])));
        }
        
        if (sb.length() == 0) return "-1";
        
        return sb.toString().startsWith("0") ? "0" : sb.toString();
    }
}
```

# 3차 풀이
```java
import java.util.*;

class Solution {
    public String solution(String X, String Y) {
        Map<Integer, Integer> xCount = new HashMap<>();
        Map<Integer, Integer> yCount = new HashMap<>();
        
        for (String x : X.split("")) {
            int target = Integer.parseInt(x);
            xCount.put(target, xCount.getOrDefault(target, 0) + 1);
        }
        
        for (String y : Y.split("")) {
            int target = Integer.parseInt(y);
            yCount.put(target, yCount.getOrDefault(target, 0) + 1);
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 9; i >= 0; i--) {
            if (!xCount.containsKey(i) || !yCount.containsKey(i)) continue;
            sb.append(String.valueOf(i).repeat(Math.min(xCount.get(i), yCount.get(i))));
        }
        
        if (sb.length() == 0) return "-1";
        return sb.toString().startsWith("0") ? "0" : sb.toString();
    }
}
```
