// 정렬, 해시
import java.util.*;

class Solution {
    public String[] solution(String[] players, String[] callings) {
        Map<String, Integer> idxByPlayer = new HashMap<>();
        for (int i = 0; i < players.length; i++) {
            idxByPlayer.put(players[i], i);
        }

        for (String target : callings) {
            swap(target, idxByPlayer, players);
        }

        return players;
    }

    private void swap(String target, Map<String, Integer> idxByPlayer, String[] players) {
        int originIdx = idxByPlayer.get(target) - 1;
        int targetIdx = idxByPlayer.get(target);

        String temp = players[originIdx];
        players[originIdx] = players[targetIdx];
        players[targetIdx] = temp;
        
        idxByPlayer.put(target, originIdx);
        idxByPlayer.put(temp, targetIdx);
    }
}

# 2차 풀이

import java.util.*;

class Solution {
    public String[] solution(String[] players, String[] callings) {
        Map<String, Integer> indexes = new HashMap<>();
        for (int i = 0; i < players.length; i++) indexes.put(players[i], i);
        for (String calling : callings) swap(calling, players, indexes);
        
        return players;
    }
    
    private void swap(String calling, String[] players, Map<String, Integer> indexes) {
        int targetIdx = indexes.get(calling);
        String temp = players[targetIdx - 1];
        players[targetIdx - 1] = players[targetIdx];
        players[targetIdx] = temp;
        
        indexes.put(players[targetIdx - 1], targetIdx - 1);
        indexes.put(players[targetIdx], targetIdx);
    }
}
