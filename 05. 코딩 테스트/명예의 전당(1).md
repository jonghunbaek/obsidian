# 1차 풀이
```java
// 우선순위 큐
import java.util.*;

class Solution {
    public int[] solution(int k, int[] score) {
        List<Integer> results = new ArrayList<>();
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int s : score) {
            pq.offer(s);

            if (pq.size() > k) {
                pq.poll();
            }

            results.add(pq.peek());
        }

        return results.stream()
                .mapToInt(Integer::intValue)
                .toArray();
    }
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(int k, int[] score) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        int[] results = new int[score.length];
        for (int i = 0; i < score.length; i++) {
            pq.offer(score[i]);
            
            if (pq.size() > k) {
                pq.poll();
            }
            
            results[i] = pq.peek();
        }
        
        return results;
    }
}
```
