# 1차 풀이
```java
import java.util.*;

class Solution {
    public int solution(int[] wallet, int[] bill) {
        Arrays.sort(wallet);
        Arrays.sort(bill);
        
        int answer = 0;
        int small = bill[0];
        int large = bill[1];
        while (small > wallet[0] || large > wallet[1]) {
            large /= 2;
            
            int temp = 0;
            if (large < small) {
                temp = small;
                small = large;
                large = temp;
            }
            
            answer++;
        }
        
        return answer;
    }
}
```
