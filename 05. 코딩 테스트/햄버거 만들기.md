# 1차 풀이
```java
public int solution(int[] ingredient) {  
    int[] hamburger = new int[] {1, 2, 3, 1};  
    int count = 0;  
    Stack<Integer> stack = new Stack<>();  
    for (int target : ingredient) {  
        if (stack.isEmpty() && target != 1) {  
            continue;  
        }  
        if (target == 1) {  
            stack.push(target);  
            continue;  
        }  
  
        if (stack.peek() == 1 && target == 2) {  
            stack.push(target);  
        } else if (stack.peek() == 2 && target == 3) {  
            stack.push(target);  
        } else if (stack.peek() == 3 && target == 1) {  
            for (int j = 0; j < 3; j++) {  
                stack.pop();  
            }  
  
            count++;  
        }  
    }  
  
    int idx = 3;  
    int answerCount = 0;  
    while (!stack.isEmpty()) {  
        int value = stack.pop();  
        if (value == hamburger[idx]) {  
            answerCount++;  
        }  
  
        if (answerCount == hamburger.length) {  
            count++;  
            answerCount = 0;  
        }  
  
        idx = idx - 1 < 0 ? 3 : idx - 1;  
    }  
  
    return count;  
}
```
