# 1차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(int[] numbers) {
        Set<Integer> results = new HashSet<>();
        for (int i = 0; i < numbers.length; i++) {
            for (int j = i + 1; j < numbers.length; j++) {
                results.add(numbers[i] + numbers[j]);
            }
        }
        
        return results.stream()
            .mapToInt(Integer::intValue)
            .sorted()
            .toArray();
    }
}
```

# 2차 풀이
```java
import java.util.*;

class Solution {
    public int[] solution(int[] numbers) {
        Set<Integer> results = new HashSet<>();
        countSum(0, 0, 0, numbers, results);
        
        return results.stream()
            .sorted()
            .mapToInt(Integer::intValue)
            .toArray();
    }
    
    private void countSum(int depth, int idx, int sum, int[] numbers, Set<Integer> results) {
        if (depth == 2) {
            results.add(sum);
            return;
        }
        
        for (int i = idx; i < numbers.length; i++) {
            countSum(depth + 1, idx + 1, sum + numbers[i], numbers, results);
        }
    }
}
```
