# 1차 풀이
```java
class Solution {
    static int result = Integer.MAX_VALUE;
    static int[] dx = new int[] {1, -1, 0, 0};
    static int[] dy = new int[] {0, 0, 1, -1};
    
    public int solution(int[][] maps) {
        boolean[][] visited = new boolean[maps.length][maps[0].length];
        dfs(1, 0, 0, maps, visited);
        return result == Integer.MAX_VALUE ? -1 : result;
    }
    
    private void dfs(int depth, int y, int x, int[][] maps, boolean[][] visited) {
        if (y == maps.length - 1 && x == maps[0].length - 1) {
            result = Math.min(result, depth);
            return;
        }
        
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            
            if (nx < 0 || ny < 0 || nx >= maps[0].length || ny >= maps.length) {
                continue;
            }
            
            if (maps[ny][nx] == 0) {
                continue;
            }
            
            if (visited[ny][nx]) {
                continue;
            }
            
            visited[ny][nx] = true;
            dfs(depth + 1, ny, nx, maps, visited);
            visited[ny][nx] = false;
        }
    }
}
```

```java
import java.util.*;

class Solution {

    static int[] dx = new int[] {1, -1, 0, 0};
    static int[] dy = new int[] {0, 0, 1, -1};
    
    public int solution(int[][] maps) {
        return bfs(1, 0, 0, maps);
    }
    
    private int bfs(int depth, int y, int x, int[][] maps) {
        boolean[][] visited = new boolean[maps.length][maps[0].length];
        Queue<Machine> q = new LinkedList<>();
        q.add(new Machine (0, 0, 1));
        
        int result = -1;
        while (!q.isEmpty()) {
            Machine current = q.poll();
            
            if (current.x == maps[0].length - 1 && current.y == maps.length - 1) {
                result = current.distance;
            }
            
            for (int i = 0; i < 4; i++) {
                int nx = current.x + dx[i];
                int ny = current.y + dy[i];

                if (nx < 0 || ny < 0 || nx >= maps[0].length || ny >= maps.length) {
                    continue;
                }

                if (maps[ny][nx] == 0) {
                    continue;
                }
                
                if (visited[ny][nx]) {
                    continue;
                }
                
                visited[ny][nx] = true;
                q.add(new Machine(nx, ny, current.distance + 1));
                
            }    
        }
        
        return result;
    }
    
    static class Machine {
        int x;
        int y;
        int distance;
        
        public Machine(int x, int y, int distance) {
            this.x = x;
            this.y = y;
            this.distance = distance;
        }
    }
}
```
